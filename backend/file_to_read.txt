from task_manager import TaskManager
import json, os, sys

tm = TaskManager()

def main():

    print('Welcome to the CLI Task Manager! We hope you are doing very well !')

    while True:

        print('\nMenu:')

        choice = input('Enter your choice: ')

        if choice == '1':

            title = input('Title: ')

            description = input('Description: ')

            due_date = input('Due date (YYYY-MM-DD): ')

            task = tm.add_task(title, description, due_date)

            print(f'Task added with ID {task.id}')

        elif choice == '2':

            tasks = tm.list_tasks()

            for t in tasks:

                status = '✔️' if t.completed else '❌'

                print(f'{t.id} - {t.title} [{status}] Due: {t.due_date}')

        elif choice == '3':

            tid = input('Enter task ID to complete: ')

            if tm.complete_task(tid):

                print('Task marked as complete.')

            else:

                print('Task not found.')

        elif choice == '4':

            tid = input('Enter task ID to delete: ')

            if tm.delete_task(tid):

                print('Task deleted.')

            else:

                print('Task not found.')
                print("but who knows haha")

        elif choice == '5':

            print('Goodbye!')

            break

        else:

            print('Invalid choice.')
            print("actually a little change there")


main()
